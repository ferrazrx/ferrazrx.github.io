          //css
          .transition {
            width: 150px;
            overflow: hidden;
            margin-left: auto;
            margin-right: auto;
          }
          .transitionImage{
            transform: translate3d(0,0,0);
            transition: transform 1s cubic-bezier(0,1.66,.83,.76);
          }
          .transitionImage:hover{
            transform: translate3d(-150px,0,0);
          }
          .toCircle {
            display: flex;
            align-items: center;
            justify-content: center;
            width: 200px;
            height: 200px;
            border-radius: 0px;
            background-color: rgba(255, 0, 0, 0.3);
            transition: all .5s ease-out;
            margin-left: auto;
            margin-right: auto;
          }
          .toCircle:hover {
            border-radius: 50%;
            background-color: rgba(255, 255, 0, 0.5);
          }
          .gray{
              -webkit-filter: grayscale(100%); /* Safari 6.0 - 9.0 */
              filter: grayscale(100%);
              transition: all .7s ease;
          }
          .gray:hover{
            cursor: pointer;
            -webkit-filter: grayscale(0%); /* Safari 6.0 - 9.0 */
            filter: grayscale(0%);
          }
          //Javascript
          var square = document.getElementsByClassName('spin')[0];
          square.addEventListener('mouseover', spin);

          function spin() {
            count = 0;
            var id = setInterval(rotate, 10);
            function rotate(){
              count++;
              square.style.transform = 'rotateY('+count+'deg)'
              if(count >= 360){
                clearInterval(id);
              }
            }
          };
